== Deploy
[source,bash]
----
NAMESPACE="msc"
kubectl apply -n $NAMESPACE -f kafka -f ksql

helm repo add kafka-ui https://provectus.github.io/kafka-ui
helm install -n $NAMESPACE -f kafka-ui/values.yml kafka-ui kafka-ui/kafka-ui
----
